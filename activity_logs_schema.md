# Activity Log Database Schema

## Create the activity_logs table

```sql
CREATE TABLE activity_logs (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    log_name VARCHAR(255) NOT NULL,
    description VARCHAR(500) NOT NULL,
    subject_type VARCHAR(255),
    subject_id BIGINT,
    event VARCHAR(255),
    causer_type VARCHAR(255),
    causer_id BIGINT,
    properties JSON,
    batch_uuid VARCHAR(255),
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_activity_logs_log_name (log_name),
    INDEX idx_activity_logs_subject (subject_type, subject_id),
    INDEX idx_activity_logs_causer (causer_type, causer_id),
    INDEX idx_activity_logs_event (event),
    INDEX idx_activity_logs_batch_uuid (batch_uuid),
    INDEX idx_activity_logs_created_at (created_at)
);
```

## Alternative for H2 Database (for testing)

```sql
CREATE TABLE activity_logs (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    log_name VARCHAR(255) NOT NULL,
    description VARCHAR(500) NOT NULL,
    subject_type VARCHAR(255),
    subject_id BIGINT,
    event VARCHAR(255),
    causer_type VARCHAR(255),
    causer_id BIGINT,
    properties CLOB,  -- H2 doesn't support JSON natively
    batch_uuid VARCHAR(255),
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_activity_logs_log_name ON activity_logs(log_name);
CREATE INDEX idx_activity_logs_subject ON activity_logs(subject_type, subject_id);
CREATE INDEX idx_activity_logs_causer ON activity_logs(causer_type, causer_id);
CREATE INDEX idx_activity_logs_event ON activity_logs(event);
CREATE INDEX idx_activity_logs_batch_uuid ON activity_logs(batch_uuid);
CREATE INDEX idx_activity_logs_created_at ON activity_logs(created_at);
```
